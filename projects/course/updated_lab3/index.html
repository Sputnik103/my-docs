    import requests
from bs4 import BeautifulSoup

def get_comments(movie_url, num_comments):
    headers = {'User-Agent': 'Mozilla/5.0'}
    response = requests.get(movie_url, headers=headers)
    soup = BeautifulSoup(response.text, 'html.parser')
    comments = soup.find_all('div', {'class': 'comment'})[:num_comments]
    return [comment.find('p').text.strip() for comment in comments]

    精确模式：试图将句子最精确地切开，适合文本分析；（不加参数，默认是精确模式）
    全模式：把句子中所有的可以成词的词语都扫描出来, 速度非常快，但是不能解决歧义（cut_all=True）
    paddle模式：利用PaddlePaddle深度学习框架，训练序列标注（双向GRU）网络模型实现分词。同时支持词性标注。（应该是百度提供的分词模型）（use_paddle=True）
    搜索引擎模式：在精确模式的基础上，对长词再次切分，提高召回率，适合用于搜索引擎分词（jieba.cut_for_search）

    Utilize the Latent Dirichlet Allocation (LDA) model to analyze the themes of movie reviews. Compare the topics discovered for each movie to gain insights into the thematic differences.

    Task Requirements:¶

    • Vectorization: Convert the preprocessed text into a document-term matrix.
    • LDA Analysis: Apply the LDA model to discover the underlying topics.
    • Result Interpretation: Discuss the significance of the discovered topics.

    Implementation Tips:¶

    • Choose an appropriate number of topics to explore.
    • Analyze the top words of each topic to interpret the themes accurately.

    from sklearn.decomposition import LatentDirichletAllocation
from sklearn.feature_extraction.text import CountVectorizer

# 定义一个函数来执行LDA主题模型分析
def lda_topic_modeling(processed_texts, num_topics):
    
    # TODO: 使用CountVectorizer对文本进行向量化
    vectorizer = CountVectorizer()
    doc_term_matrix = vectorizer.fit_transform(processed_texts)
    #

    # TODO: 使用LatentDirichletAllocation进行主题分析
    lda = LatentDirichletAllocation(n_components=num_topics, random_state=0)
    lda.fit(doc_term_matrix)
    #

    # 获取每个主题的词分布
    topic_words = {}
    words = vectorizer.get_feature_names_out()
    
    for topic, comp in enumerate(lda.components_):
        word_idx = comp.argsort()[-5:]  # 每个主题取权重最大的5个词
        topic_words[topic] = [words[i] for i in word_idx]
    return topic_words
